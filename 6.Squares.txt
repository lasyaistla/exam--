<script>
    function inputnum() {
        let l = prompt("Enter the limit")
        let j, result
        for (j = 1; j <= l; j++) {
            result = j * j;
            document.write("Square of " + j + " is " +
                result)
            document.write("</br>")
        }
    }
</script>
