    <script>
        function addfunc() {
            let num1 = prompt("Enter First Number");
            let num2 = prompt("Enter Second Number");
            num1 = Number(num1);
            num2 = Number(num2);
            let res = num1 + num2;
            alert("Sum of " + num1 + " and " + num2 + " is " + res);
        }
    </script>
