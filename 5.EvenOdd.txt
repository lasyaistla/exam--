        <input type="number" id="ip">
        <input type="submit" onclick="evenodd()"></form>
    <script>
        function
        evenodd() {
            let num = document.getElementById("ip").value
            if (num % 2 == 0) {
                alert(num + " is an Even Number");
            } else {
                alert(num+ " is an Odd Number");
            }
        }
    </script>
